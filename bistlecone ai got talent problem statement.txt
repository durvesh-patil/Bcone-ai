class NewsMonitor:

        """Fetch latest news affecting supply chains"""
        url = f"https://newsapi.org/v2/everything"
        params = {
            'q': ' OR '.join(self.keywords),
            'language': 'en',
            'sortBy': 'publishedAt',
            'pageSize': 20,
            'apiKey': self.news_api_key
        }
        response = requests.get(url, params=params)
        return response.json()['articles']
    
    def get_location_mentions(self, article):
        """Extract countries/ports/regions from article"""
        # Use NER (Named Entity Recognition) or simple keyword matching

from datetime import datetime

class RiskAwareOptimizer:
    def __init__(self):
        self.news_monitor = NewsMonitor()
        self.ai_agent = SupplyChainRiskAgent()
        self.risk_penalties = {
            'LOW': 1.1,      # 10% cost penalty
            'MEDIUM': 1.3,   # 30% cost penalty
            'HIGH': 1.6,     # 60% cost penalty
            'CRITICAL': 99   # Effectively blocked
        }
    
    async def optimize_with_risk_awareness(self, route_request):
        """Main optimization considering real-time risks"""
        
        # 1. Get latest news
        news_articles = self.news_monitor.fetch_supply_chain_news()
        
        # 2. Analyze impact on proposed route
        risks = []
        for article in news_articles:
            analysis = await self.ai_agent.analyze_news_impact(
                article, 
                route_request
            )
            if analysis['affects_route']:
                risks.append(analysis)
        
        # 3. Calculate risk-adjusted costs
        routes = self.calculate_base_routes(route_request)
        
        for route in routes:
            route['base_cost'] = route['cost']
            route['base_time'] = route['time']
            
            # Apply risk penalties
            applicable_risks = self.get_risks_for_route(route, risks)
            max_severity = max([r['severity'] for r in applicable_risks], 
                              default='LOW')
            
            route['risk_score'] = max_severity
            route['risk_adjusted_cost'] = (
                route['cost'] * self.risk_penalties[max_severity]
            )
            route['active_risks'] = applicable_risks
        
        # 4. Sort by risk-adjusted cost
        routes.sort(key=lambda x: x['risk_adjusted_cost'])
        
        return {
            'recommended_route': routes[0],

            'active_risks': risks,

### **Core Components**

#### **Component 1: Intelligent Route Optimization Engine**

**What It Does:**
- Calculates optimal transport routes across **4 modes**: Sea, Air, Rail, Truck
- Supports **multi-leg journeys** (e.g., Truck ‚Üí Port ‚Üí Sea ‚Üí Rail ‚Üí Final delivery)
- Optimizes for **3 priorities**: Cost-first, Time-first, or Balanced
- Handles **real-world constraints**: cargo weight, delivery deadlines, budget limits, carrier capacity
- Provides **top 3 alternatives** with detailed trade-off analysis

**How It Works:**
```
User Input: Shanghai ‚Üí Hamburg, 5000kg electronics, deliver by Dec 15
         ‚Üì
Algorithm evaluates all possible combinations:
  - Pure modes (all-sea, all-air, all-rail)
  - Multi-modal combinations (truck+sea+rail)
  - Different carriers and routes
         ‚Üì
Scoring based on weighted factors:
  - Cost (base rate + fuel + handling + customs)
  - Time (transit + port dwell + customs clearance)
  - Risk (geopolitical stability, weather, congestion)
  - Carbon footprint (ESG compliance)
         ‚Üì
Output: Top 3 options with visual comparison
  Option A: Rail-dominant - $4,300, 13 days, Medium risk
  Option B: Sea via Suez - $4,000, 21 days, High risk (Red Sea)
  Option C: Air freight - $15,000, 2 days, Low risk
```

**Technical Implementation:**
- **Graph-based routing algorithm** with weighted edges
- **Dynamic cost modeling** (fuel prices, seasonal rates, demand-based pricing)
- **Constraint satisfaction** (delivery deadlines, cargo restrictions)
- **Heuristic optimization** for complex multi-leg scenarios

**Data Sources:**
- Transport network database (ports, airports, rail terminals, trucking hubs)

**What It Does:**

- **Calculates specific impact** for each affected route (cost increase, time delay)

  Alternative 1: Shanghai ‚Üí Rotterdam via Cape of Good Hope
